# Kill sessions
psql -U postgres -c "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname='cse4308_12345' AND pid <> pg_backend_pid();"
psql -U postgres -c "DROP DATABASE IF EXISTS cse4308_12345;"
psql -U postgres -c "DROP ROLE IF EXISTS alice;"

# Check DB
$DB_CHECK = psql -U postgres -tAc "SELECT 1 FROM pg_database WHERE datname='cse4308_12345';"
if (-not $DB_CHECK) {
    Write-Host "Database cse4308_12345 does not exist (removed)."
} else {
    Write-Host "Database cse4308_12345 still exists."
}

# Check role
$USER_CHECK = psql -U postgres -tAc "SELECT 1 FROM pg_roles WHERE rolname='alice';"
if (-not $USER_CHECK) {
    Write-Host "Role alice does not exist (removed)."
} else {
    Write-Host "Role alice still exists."
}
