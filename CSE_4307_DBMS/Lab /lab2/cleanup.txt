psql -U postgres -c "SELECT pg_terminate_backend(pid) FROM pg_stat_activity WHERE datname='cse4308_12345';"
psql -U postgres -c "DROP DATABASE IF EXISTS cse4308_12345;"
psql -U postgres -c "DROP ROLE IF EXISTS alice;"





DB_CHECK=$(psql -U postgres -tAc "SELECT 1 FROM pg_database WHERE datname='cse4308_12345';")
if [ -z "$DB_CHECK" ]; then
    echo "Database cse4308_12345 does not exist (removed)."
else
    echo "Database cse4308_12345 still exists."
fi

USER_CHECK=$(psql -U postgres -tAc "SELECT 1 FROM pg_roles WHERE rolname='alice';")
if [ -z "$USER_CHECK" ]; then
    echo "Role alice does not exist (removed)."
else
    echo "Role alice still exists."
fi
